import { defineComponent } from 'vue';
import { BasicTable, useTable } from '/@/components/Table';
import { collectColumns } from './daily.data';
import { getWeekDataList } from '/@/utils/validate';

import dayjs from 'dayjs';


export default defineComponent({
  name: '',
  components: {
    BasicTable,
  },
  setup() {
    // 注册表格
    const [registerTable, { reload }] = useTable({
      title: '汇总表',
      rowKey: 'dailyDetailId',
      striped: false,
      columns: collectColumns,
      afterFetch: (table) => {
        table = arr;
        const tableFormatter = new TableFommatter(table);
        tableFormatter
          .dateFormatter('YYYY-MM-DD')
          .sortByDate()
          .recursiveSort(['dailyDate'], 'combineInfo', 'orgPleatName')
          .recursiveSort(['dailyDate', 'orgPleatName'], 'pleatCombineInfo', 'orgName')
          .recursiveSort(
            ['dailyDate', 'orgPleatName', 'orgName'],
            'orgCombineInfo',
            'journalTypeName',
          )
          .recursiveSort(
            ['dailyDate', 'orgPleatName', 'orgName', 'journalTypeName'],
            'typeCombineInfo',
            'bankName',
          )
          .recursiveSort(
            ['dailyDate', 'orgPleatName', 'orgName', 'journalTypeName', 'bankName'],
            'bankCombineInfo',
            null,
          );
        return tableFormatter.table;
      },
      formConfig: {
        labelWidth: 80,
        schemas: searchCollectFormSchema,
        showResetButton: hasPermission(['fund:daily:get']),
        showSubmitButton: hasPermission(['fund:daily:get']),
      },
      showIndexColumn: false,
      useSearchForm: true,
      showTableSetting: true,
      bordered: true,
      immediate: true,
      beforeFetch: (table) => {
        let currentDate = new Date();
        let year = currentDate.getFullYear(); //获取年份
        let month = currentDate.getMonth() + 1; ///获取月份 月份是从0开始的，所以要加1
        let date = currentDate.getDate();
        date = date < 10 ? '0' + date : date;
        if (table.time == 'day') {
          table.startTime = year + '-' + month + '-' + date;
          table.endTime = year + '-' + month + '-' + date;
        } else if (table.time == 'week') {
          // 获取当前时间这一周的日期
          let weekDate = getWeekDataList(year + '-' + month + '-' + date);
          table.startTime = weekDate[0];
          table.endTime = weekDate[weekDate.length - 1];
        } else if (table.time == 'moment') {
          let lastDate = new Date(year, month, 0).getDate(); //获取当月最后一日
          table.startTime = year + '-' + month + '-' + '01';
          table.endTime = year + '-' + month + '-' + lastDate;
        } 
      },
    });

    interface TableFormatter {
      table: any;
    }
    class TableFormatter {
      constructor(table) {
        this.table = table;
      }
      // 日期格式化
      dateFormatter(formatter) {
        this.table.forEach((v) => (v.dailyDate = dayjs(v.dailyDate).format(formatter)));
        return this;
      }
      // 按日期排序-从大到小
      sortByDate() {
        this.table.sort((a, b) => dayjs(b.dailyDate).unix() - dayjs(a.dailyDate).unix());
        return this;
      }
      // 表格数据依次排序并添加合并信息
      recursiveSort(recordKeysArr, combineInfo, nextSortKey) {

        const separate = this.table.reduce((pre, cur, k) => {
          const recordKey = recordKeysArr.map((v) => cur[v]).join('-');
          if (!pre[recordKey]) {
            pre[recordKey] = { part: [cur], [combineInfo]: { startIdx: k, rowSpan: 1 } };
          } else {
            pre[recordKey].part.push(cur);
            pre[recordKey][combineInfo].rowSpan += 1;
          }
          cur[combineInfo] = pre[recordKey][combineInfo];
          return pre;
        }, {});
        if (nextSortKey) {
          this.table = Object.keys(separate).reduce((pre, cur) => {
            separate[cur].part.sort((a, b) => {
              if (a[nextSortKey] && b[nextSortKey]) {
                return a[nextSortKey].localeCompare(b[nextSortKey]);
              } else {
                return 0;
              }
            });
            return pre.concat(separate[cur].part);
          }, []);
        }
        return this;
      }
    }


    return {
      registerTable,
    };
  },
});

getWeekDataList 方法

// 获取当前日期的这一周
export function getWeekDataList(data: any) {
  //根据日期获取本周周一~周日的年-月-日
  let weekList: any[] = [];
  let date: any = new Date(data);
  //判断本日期是否为周日，获取本周一日期
  if (date.getDay() == '0') {
    date.setDate(date.getDate() - 6);
  } else {
    date.setDate(date.getDate() - date.getDay() + 1);
  }
  var myDate = date.getDate();
  var myMonth = date.getMonth() + 1;
  if (date.getDate() < 10) {
    myDate = '0' + myDate;
  }
  if (date.getMonth() + 1 < 10) {
    myMonth = '0' + myMonth;
  }
  weekList.push(date.getFullYear() + '-' + myMonth + '-' + myDate);
  // 获取周二以后日期
  for (var i = 0; i < 6; i++) {
    date.setDate(date.getDate() + 1);
    myDate = date.getDate();
    myMonth = date.getMonth() + 1;
    if (date.getDate() < 10) {
      myDate = '0' + myDate;
    }
    if (date.getMonth() + 1 < 10) {
      myMonth = '0' + myMonth;
    }
    weekList.push(date.getFullYear() + '-' + myMonth + '-' + myDate);
  }
  return weekList;
}
