const status = ref("more"); // more | loading | noMore

  page: 1,
  rows: 10,
  totalCount: 0,

function handleSearch() {
  pageInfo.value = {
    page: 1,
    rows: 10,
    totalCount: 0,
  };
  getData(active.value, contractName.value, true);
}

function handleSearchConfirm() {
  pageInfo.value = {
    page: 1,
    rows: 10,
    totalCount: 0,
  };
  getData(active.value, contractName.value, true);
}
// 公共时间格式化方法
const formatContractDates = (contracts: any[]) => {
  return contracts?.forEach((item: any) => {
    item.gmtCreate = dayjs(Number(item.gmtCreate)).format("YYYY-MM-DD");
  });
};

// 公共数据获取模板
async function fetchListData(apiFn: Function, searchKey?: string, filter?: Function) {
  try {
    let hasEnoughData = false;

    // 新增递归加载方法
    const loadPages = async (): Promise<boolean> => {
      const res = await apiFn({ ...pageInfo.value, searchKey });
      if (res.code !== 1) return false;

      const rawData = res.data.data;
      const newData = filter ? rawData.filter(filter) : rawData;
      const isFullPage = rawData.length >= pageInfo.value.rows;

      // 合并数据
      contracts.value = pageInfo.value.page === 1
        ? newData
        : [...contracts.value, ...newData];

      formatContractDates(newData);

      // 判断是否满足数据量要求
      hasEnoughData = newData.length > 6;

      // 更新分页信息
      pageInfo.value.page += isFullPage ? 1 : 0;
      status.value = isFullPage ? "more" : "noMore";

      return isFullPage && !hasEnoughData;
    };

    // 循环加载直到满足条件
    let shouldContinue = true;
    while (shouldContinue) {
      shouldContinue = await loadPages();
    }
  } catch (error) {
    // 错误处理保持原样
    status.value = "noMore";
  } finally {
  }
}

// 获取待我签署列表
async function getPendingList(searchKey?: string) {
  await fetchListData(whenISignPageList, searchKey);
}

// 获取已签署列表 
async function getContractSuccess(searchKey?: string) {

}

// 获取待他人签署列表
async function getContractByOther(searchKey?: string) {
  await fetchListData(signByOthersPageList, searchKey);
}

// 获取抄送列表
async function getContractCC(searchKey?: string) {
  await fetchListData(contractCc, searchKey);
}

function handleMenuFalse() {
  appStore.setDropdownStatus(false);
}

function handleAccountChange() {
  pageInfo.value = {
    page: 1,
    rows: 10,
    totalCount: 0,

  getData(active.value, undefined, true);
}

async function getData(val: any, contractName?: string, loadingShow?: boolean) {
  loading.value = loadingShow ? true : false;
  switch (val) {
    case 1:
      await getPendingList(contractName);
      break;
    case 5:
      await getContractSuccess(contractName);
      break;
    case 4:
      await getContractByOther(contractName);
      break;
    case 2:
      await getContractCC(contractName);
      break;
  }
  loading.value = false;
}

watch(active, (newVal, oldVal) => {
  pageInfo.value = {
    page: 1,
    rows: 10,
    totalCount: 0,
  };
  getData(newVal, undefined, true);
});

onReachBottom(() => {
  console.log("onReachBottom", status.value);
  if (status.value === "more" && !loading.value) {
    status.value = "loading";
    getData(active.value, contractName.value, false);
  }
});

onPullDownRefresh(() => {
  console.log("refresh");
  timer.value = setTimeout(async () => {
    pageInfo.value = {
      page: 1,
      rows: 10,
      totalCount: 0,
    };
    await getData(active.value, contractName.value, true);
    uni.stopPullDownRefresh();
  }, 1000);
});
onShow(async () => {
  try {
    await getData(active.value, undefined, true);
  } catch (error) {
  }

});
onUnmounted(() => {
  clearTimeout(timer.value);
  timer.value = null;
});

方案一：高阶函数封装（函数式编程）

// 创建分页处理器（高阶函数）
const createPaginationHandler = (apiFn: Function, filter?: Function) => {
  return async (searchKey?: string) => {
    try {
      let currentPage = pageInfo.value.page;
      let shouldContinue = true;
      
      while (shouldContinue) {
        const res = await apiFn({ 
          page: currentPage,
          rows: pageInfo.value.rows,
          searchKey 
        });

        if (res.code === 1) {
          const rawData = res.data.data;
          const newData = filter ? rawData.filter(filter) : rawData;
          const isFullPage = rawData.length >= pageInfo.value.rows;

          // 合并数据
          contracts.value = currentPage === 1 
            ? newData 
            : [...contracts.value, ...newData];
          
          formatContractDates(newData);
          
          // 更新状态
          shouldContinue = isFullPage && newData.length === 0;
          currentPage += isFullPage ? 1 : 0;
          pageInfo.value.page = currentPage;
          status.value = isFullPage ? "more" : "noMore";
        } else {
          shouldContinue = false;
        }
      }
    } catch (error) {
      status.value = "noMore";
    }
  }
}

// 业务方法改造
const getPendingList = createPaginationHandler(whenISignPageList);
const getContractSuccess = createPaginationHandler(
  allContractPageList, 
  (item: any) => item.state == 2
);

生成器方案按需加载，内存友好大数据量分页场景

保持最佳性能（无额外对象创建开销）
